## 🗄️ Alterações no Banco de Dados

### Migration: `v1.1.0-streak-preferences.sql`

```sql
-- 1. Adicionar campos de streak
ALTER TABLE choppinho.users
ADD COLUMN IF NOT EXISTS current_streak INTEGER DEFAULT 0,
ADD COLUMN IF NOT EXISTS longest_streak INTEGER DEFAULT 0,
ADD COLUMN IF NOT EXISTS last_activity_week VARCHAR(7);

-- 2. Função para calcular streak
CREATE OR REPLACE FUNCTION choppinho.calculate_weekly_streak(p_user_id UUID)
RETURNS TABLE(current_streak INTEGER, longest_streak INTEGER) AS $$
DECLARE
  v_weeks TEXT[];
  v_current INTEGER := 0;
  v_longest INTEGER := 0;
  v_temp INTEGER := 0;
  v_last_week TEXT;
  v_this_week TEXT;
BEGIN
  -- Buscar semanas com atividades (formato ISO: 2026-W08)
  SELECT ARRAY_AGG(DISTINCT TO_CHAR(start_date, 'IYYY-IW') ORDER BY TO_CHAR(start_date, 'IYYY-IW') DESC)
  INTO v_weeks
  FROM choppinho.activities
  WHERE user_id = p_user_id;

  -- Calcular streak atual
  v_this_week := TO_CHAR(NOW(), 'IYYY-IW');
  v_last_week := v_weeks[1];

  -- Lógica de streak consecutivo...
  -- (simplificado aqui, implementar completo na migration)

  RETURN QUERY SELECT v_current, v_longest;
END;
$$ LANGUAGE plpgsql;

-- 3. Trigger para atualizar streak automaticamente
CREATE OR REPLACE FUNCTION choppinho.trigger_update_streak()
RETURNS TRIGGER AS $$
BEGIN
  -- Atualizar streak do usuário após insert de atividade
  UPDATE choppinho.users
  SET current_streak = (SELECT current_streak FROM choppinho.calculate_weekly_streak(NEW.user_id)),
      longest_streak = GREATEST(longest_streak, (SELECT longest_streak FROM choppinho.calculate_weekly_streak(NEW.user_id)))
  WHERE id = NEW.user_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_activity_insert
AFTER INSERT ON choppinho.activities
FOR EACH ROW
EXECUTE FUNCTION choppinho.trigger_update_streak();

-- 4. Índices para performance
CREATE INDEX IF NOT EXISTS idx_activities_user_week
ON choppinho.activities (user_id, start_date);
```
